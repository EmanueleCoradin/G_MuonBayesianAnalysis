
model {
  for (i in 1:N) {
    t_data[i] ~ dnorm(mu[i], tau_y)
    mu[i] <- exp(-t[i] / tau_fixed) * (1 + a * cos(b * t[i] + c))
  }

  # Priors
  a ~ dunif(-1/3, 1)
  b ~ dunif(4, 6)
  c ~ dunif(-3.14/2, 3.14/2)
  tau_y ~ dgamma(0.001, 0.001) # Precision of the normal distribution (inverse of variance)
}
